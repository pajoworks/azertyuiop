DOC TECHNIQUE

NEXUS RP NETWORK // PLATEFORME COLLABORATEURS // v2.4.1

Ce document décrit l'ensemble du code de la plateforme NEXUS RP, conçue pour permettre aux collaborateurs d'un serveur GTA RP de se connecter avec un compte personnel, gérer leur profil, consulter l'annuaire des membres, et accéder au règlement.

🧠 Technologie : HTML5 + CSS3 + JavaScript pur (Vanilla JS). Aucun framework externe requis. La base de données utilise le localStorage du navigateur — parfait pour un hébergement statique type GitHub Pages (comme le site CHL original).

HTML5 CSS3 VANILLA JS LOCALSTORAGE DB SESSIONS DARK UI RESPONSIVE

ARCHITECTURE

La plateforme fonctionne entièrement côté client (frontend-only). Tout le code est dans un seul fichier index.html contenant les blocs HTML, CSS et JavaScript.

Flux général

Utilisateur ouvre le site
       │
       ├─ Session active dans sessionStorage ?
       │     ├─ OUI → affiche le Dashboard directement
       │     └─ NON → affiche la page Login
       │
 [Page Login]
       │
       ├─ Onglet CONNEXION → handleLogin()
       │     ├─ Vérifie username + password (vs localStorage)
       │     ├─ Crée une session (sessionStorage)
       │     └─ showDashboard()
       │
       └─ Onglet INSCRIPTION → handleRegister()
             ├─ Valide les champs
             ├─ Crée un objet user dans localStorage
             └─ Message de succès → redirection vers Connexion

 [Dashboard]
       │
       ├─ /home      → stats + annonces
       ├─ /profil    → fiche perso + formulaire edit
       ├─ /membres   → tableau + recherche + filtres
       ├─ /reglement → accordion règles
       └─ /admin     → (admin only) gestion comptes

BASE DE DONNÉES — API

L'objet DB est le seul point d'accès aux données. Toutes les opérations passent par lui.

const DB = {
  // Lire tous les users
  getUsers() {
    return JSON.parse(localStorage.getItem('nexus_users') || '[]');
  },
  // Sauvegarder le tableau complet
  saveUsers(users) {
    localStorage.setItem('nexus_users', JSON.stringify(users));
  },
  // Trouver par username
  getUserByUsername(username) {
    return this.getUsers()
      .find(u => u.username.toLowerCase() === username.toLowerCase());
  },
  // Créer un nouveau compte
  createUser(data) {
    const users = this.getUsers();
    const user = {
      id:        'USR' + Date.now().toString(36).toUpperCase(),
      username:  data.username,
      password:  btoa(data.password),  // encodage base64
      ic:        data.ic || '',
      faction:   data.faction || 'Civil',
      discord:   data.discord || '',
      role:      'membre',
      bio:       '',
      createdAt: new Date().toLocaleDateString('fr-FR')
    };
    users.push(user);
    this.saveUsers(users);
    return user;
  },
  // Mettre à jour un user (merge partiel)
  updateUser(username, updates) {
    const users = this.getUsers();
    const idx = users.findIndex(
      u => u.username.toLowerCase() === username.toLowerCase()
    );
    if (idx === -1) return false;
    users[idx] = { ...users[idx], ...updates };
    this.saveUsers(users);
    return users[idx];
  },
  // Supprimer un user
  deleteUser(username) {
    let users = this.getUsers()
      .filter(u => u.username.toLowerCase() !== username.toLowerCase());
    this.saveUsers(users);
  }
};

BASE DE DONNÉES — SESSIONS

La session utilise sessionStorage (pas localStorage) afin que la session expire automatiquement à la fermeture de l'onglet. 

// Créer une session (à la connexion)
DB.setSession(user) {
  sessionStorage.setItem('nexus_session', JSON.stringify(user));
}

// Lire la session
DB.getSession() {
  return JSON.parse(sessionStorage.getItem('nexus_session') || 'null');
}

// Détruire la session (déconnexion)
DB.clearSession() {
  sessionStorage.removeItem('nexus_session');
}

Pour une vraie sécurité — Backend recommandé

// Backend Node.js + Express (exemple)
const bcrypt = require('bcrypt');
const SALT   = 12;

// Inscription
app.post('/register', async (req, res) => {
  const hash = await bcrypt.hash(req.body.password, SALT);
  await db.query('INSERT INTO users (...) VALUES (...)', [hash, ...]);
});

// Connexion
app.post('/login', async (req, res) => {
  const user = await db.query('SELECT * FROM users WHERE username=?');
  const ok   = await bcrypt.compare(req.body.password, user.password);
  if (ok) req.session.userId = user.id;
});

JAVASCRIPT — AUTHENTIFICATION

Connexion — handleLogin()

function handleLogin() {
  const username = document.getElementById('login-username').value.trim();
  const password = document.getElementById('login-password').value;

  // 1. Validation des champs
  if (!username || !password) return showAlert(errEl, '⚠ Champs vides', 'error');

  // 2. Vérification contre la base de données
  const user = DB.getUserByUsername(username);
  if (!user || atob(user.password) !== password) {
    return showAlert(errEl, '✕ IDENTIFIANTS INCORRECTS', 'error');
  }

  // 3. Création session + redirection
  DB.setSession(user);
  currentUser = user;
  showDashboard();
}

Inscription — handleRegister()

function handleRegister() {
  // Validations successives :
  // ✓ Champs obligatoires (username, password, faction)
  // ✓ Longueur username (min 3 chars)
  // ✓ Correspondance passwords
  // ✓ Longueur password (min 6 chars)
  // ✓ Username déjà pris (DB.getUserByUsername)

  if (DB.getUserByUsername(username))
    return showAlert(errEl, '✕ Ce pseudo est déjà utilisé.', 'error');

  DB.createUser({ username, ic, password, faction, discord });
  showAlert(sucEl, '✔ Compte créé !', 'success');
}

Déconnexion — handleLogout()

function handleLogout() {
  DB.clearSession();        // vide sessionStorage
  currentUser = null;       // reset variable globale
  showPage('page-login');   // retour login
  showToast('Session fermée.');
}

function loadProfilePage() {

  document.getElementById('psr-ic').textContent     = u.ic      || '—';

  // Couleur du badge selon le rôle

  roleBadge.style.color = `var(--neon-${roleColors[u.role]})`;

  // Pré-remplir le formulaire d'édition
  document.getElementById('edit-ic').value = u.ic || '';
}

// Sauvegarder les modifications
function saveProfile() {
  const ic      = document.getElementById('edit-ic').value.trim();
  const faction = document.getElementById('edit-faction').value;
  const discord = document.getElementById('edit-discord').value.trim();
  const bio     = document.getElementById('edit-bio').value;

  // Mettre à jour en DB
  DB.updateUser(currentUser.username, { ic, faction, discord, bio });

  // Rafraîchir le currentUser depuis la DB
  currentUser = DB.getUserByUsername(currentUser.username);
  DB.setSession(currentUser);

  loadProfilePage();
  showToast('✔ Profil sauvegardé.');
}

// Changer le mot de passe
function changePassword() {
  const oldPw = document.getElementById('old-pw').value;
  const newPw = document.getElementById('new-pw').value;
  const u = DB.getUserByUsername(currentUser.username);

  // Vérifier l'ancien mot de passe
  if (atob(u.password) !== oldPw) return showAlert(errEl, '✕ Ancien MDP incorrect', 'error');
  if (newPw.length < 6) return showAlert(errEl, '⚠ Trop court', 'error');

  DB.updateUser(currentUser.username, { password: btoa(newPw) });
  showToast('✔ Mot de passe changé.');
}

// Générer le tableau des membres
function renderMembers() {
  const query = document.getElementById('members-search').value.toLowerCase();
  let users = DB.getUsers();

  // Filtre faction
  if (factionFilter) users = users.filter(u => u.faction?.includes(factionFilter));

  // Filtre recherche textuelle
  if (query) users = users.filter(u =>
    u.username.toLowerCase().includes(query) ||
    (u.ic||'').toLowerCase().includes(query)
  );

  // Générer les lignes HTML
  const tbody = document.getElementById('members-tbody');
  tbody.innerHTML = '';
  users.forEach(u => {
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td>...avatar + username...</td>
      <td>${escHtml(u.ic||'—')}</td>
      <td>${escHtml(u.faction||'—')}</td>
      <td>...badge rôle coloré...</td>
    `;
    tbody.appendChild(tr);
  });
}

function renderAdminUsers() {
